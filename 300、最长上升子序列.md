### 思路解析

动态规划，时间复杂度$0(n^2)$

```python

class Solution:
    def lengthOfLIS(self, array):
        if not array:
            return 0
        # 存储以nums[i]结尾的最长递增子序列的长度
        dp = [1] * len(array)
        for i in range(len(array)):
            for j in range(i):
                if array[j] < array[i]:
                    dp[i] = max(dp[i], dp[j] + 1)
        return max(dp)


```

import bisect
class Solution:
    def lengthOfLIS(self, array):
        if not array:
            return 0

        dp = []
        for num in array:
            index = bisect.bisect_left(dp, num)
            if index < len(dp):
                dp[index] = num
            else:
                dp.append(num)

        return len(dp)

```

class Solution:
    def lengthOfLIS(self, array):
        if not array:
            return 0

        dp = [array[0]]
        for num in array[1:]:
            index = self.binarySearch(dp, num)
            if index < len(dp):
                dp[index] = num
            else:
                dp.append(num)

        return len(dp)

    def binarySearch(self, array, num):
        left, right = 0, len(array)
        while left < right:
            mid = (left + right) // 2
            if array[mid] < num:
                left = mid + 1
            else:
                right = mid
        return left


```

#### 输出最长递增子序列

```python

class Solution:
    def lengthOfLIS(self, array):
        if not array:
            return 0

        # 存储以nums[i]结尾的最长递增子序列的长度
        dp = [1] * len(array)
        for i in range(1, len(array)):
            for j in range(i):
                if array[j] < array[i]:
                    dp[i] = max(dp[i], dp[j] + 1)
        return max(dp), dp

    def printOneLIS(self, array):
        maxLen, dp = self.lengthOfLIS(array)
        res, index = [None] * maxLen, maxLen - 1
        for i in range(len(array) - 1, -1, -1):
            if dp[i] == maxLen:
                res[index] = array[i]
                maxLen -= 1
                index -= 1
        return res

```

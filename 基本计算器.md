### 思路解析

递归法
```python

class Solution:
    def calculate(self, s: str) -> int:
        return self.calculateHelper(s, 0)[0]

    def calculateHelper(self, s, start):
        pre = 0
        que = []
        while start < len(s) and s[start] != ')':
            if s[start] >= '0' and s[start] <= '9':
                pre = 10 * pre + int(s[start])
            elif s[start] != '(':
                que.append(pre)
                que.append(s[start])
                pre = 0
            else:
                pre, start = self.calculateHelper(s, start + 1)
            start += 1
        que.append(pre)
        return self.addNum(que), start

    def addNum(self, que):
        res = 0
        sign = 1
        while que:
            num = que.pop(0)
            if num == '+':
                sign = 1
            elif num == '-':
                sign = -1
            else:
                res += sign * num
        return res


```
非递归法

```python

class Solution:
    def calculate(self, s: str) -> int:
        num, sign, res = 0, 1, 0
        stack = []
        for char in s:
            if char.isdigit():
                num = 10 * num + int(char)
            elif char in {'+', '-'}:
                res += sign * num
                num = 0
                sign = [-1, 1][char == '+']
            elif char == '(':
                stack.append(res)
                stack.append(sign)
                res = 0
                sign = 1
            elif char == ')':
                res += sign * num
                num = 0
                res *= stack.pop()
                res += stack.pop()

        return res + sign * num

```

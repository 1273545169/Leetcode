### 思路解析

```python

from collections import deque

class MyStack:
    def __init__(self):
        self.data = deque()
        self.help = deque()

    def push(self, x: int) -> None:
        if not self.data:
            self.data.append(x)
        else:
            self.help.append(x)
            while self.data:
                self.help.append(self.data.popleft())

            while self.help:
                self.data.append(self.help.popleft())

    def pop(self) -> int:
        if self.data:
            return self.data.popleft()

    def top(self) -> int:
        if self.data:
            top = self.data.popleft()
            self.data.appendleft(top)
            return top

    def empty(self) -> bool:
        return False if self.data else True

```
